cmake_minimum_required(VERSION 3.12)

# Pull in SDK (must be before project)
include(pico_sdk_import.cmake)

project(display_test C CXX ASM)
set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

# Initialize the SDK
pico_sdk_init()

add_executable(display_test
    test_display.c
    src/pico/lcd.c
)

target_include_directories(display_test PRIVATE
    ${CMAKE_CURRENT_LIST_DIR}/src/pico
)

target_compile_definitions(display_test PRIVATE
    PICO_CORE1_STACK_SIZE=256
)

target_link_libraries(display_test
    pico_stdlib
    hardware_pio
    hardware_pwm
    hardware_dma
    hardware_clocks
)

pico_add_extra_outputs(display_test)

# Enable stdio (no USB, just UART if available)
pico_enable_stdio_usb(display_test 0)
pico_enable_stdio_uart(display_test 0)
